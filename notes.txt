function fn(){
    console.log('fn is invoked')

function spinnerFactory(){
    var counter = 0;

    function increment(){
        return ++counter;
    }

    function decrement(){
        return --counter;
    }

    return {
        increment: increment,
        decrement: decrement
    }    
}

//usage: 
var spinner = spinnerFactory()
spinner.increment() //=> 1
spinner.increment() //=> 2
spinner.increment() //=> 3

spinner.decrement() //=> 2
spinner.decrement() //=> 1
spinner.decrement() //=> 0
